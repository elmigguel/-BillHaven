## 10. React Component Boilerplate

### Quick Logo Component
```tsx
// src/components/CryptoLogo.tsx
import React from 'react';

interface CryptoLogoProps {
  name: 'polygon' | 'ethereum' | 'arbitrum' | 'optimism' | 'base' | 'bsc' | 'bitcoin' | 'solana' | 'ton';
  size?: number;
}

const LOGO_URLS = {
  polygon: 'https://cryptologos.cc/logos/polygon-matic-logo.svg',
  ethereum: 'https://cryptologos.cc/logos/ethereum-eth-logo.svg',
  arbitrum: 'https://cryptologos.cc/logos/arbitrum-arb-logo.svg',
  optimism: 'https://cryptologos.cc/logos/optimism-ethereum-op-logo.svg',
  base: '/assets/chains/base.svg',
  bsc: 'https://cryptologos.cc/logos/bnb-bnb-logo.svg',
  bitcoin: 'https://cryptologos.cc/logos/bitcoin-btc-logo.svg',
  solana: 'https://cryptologos.cc/logos/solana-sol-logo.svg',
  ton: 'https://ton.org/download/ton_symbol.svg'
};

export const CryptoLogo: React.FC<CryptoLogoProps> = ({ name, size = 32 }) => {
  return (
    <img
      src={LOGO_URLS[name]}
      alt={name}
      width={size}
      height={size}
      style={{ borderRadius: '50%' }}
    />
  );
};

// Usage:
<CryptoLogo name="polygon" size={40} />
```
